# Create subset of data at target syllable offset (time_zero == 4)
# and calculate mean/se of target fixations for each group in each 
# stress condition
bi_4_prop <- stress_50 %>% 
  filter(group != "Monolingual", 
    time_zero == 4) %>% 
  group_by(id, group, stress, use_z, prof_z) %>% 
  summarize(avg = mean(target_prop), se = std_err(target_prop), 
    test_max = avg + se, .groups = "drop") %>% 
  mutate(group_sum = if_else(group == "HL", -1, 1), 
    stress_sum = if_else(stress == "Paroxytone", -1, 1))

# Model formula
beta_reg_formula <- bf(
  family = zero_inflated_beta(
    link = "logit", 
    link_phi = "log", 
    link_zi = "logit"), 
  avg ~ 1 + 
    group_sum * stress_sum * use_z * prof_z + 
    (1 + stress_sum | id), 
  phi ~ 1 + (1 | id), 
  zi ~ 1 + (1 | id)
)

# Set priors
beta_reg_prior <- c(
  prior(normal(0, 0.5), class = "Intercept"), 
  prior(normal(0, 0.5), class = "Intercept", dpar = "phi"), 
  prior(logistic(0, 1), class = "Intercept", dpar = "zi"), 
  prior(normal(0, 0.2), class = "b"), 
  prior(student_t(3, 0, 2.5), class = "sd"), 
  prior(lkj(1), class = "cor")
)

# Sample priors
beta_reg_prior_predictive_check <- brm(
  formula = beta_reg_formula, 
  prior = beta_reg_prior, 
  data = bi_4_prop, 
  cores = 4, backend = "cmdstanr", 
  sample_prior = "only", 
  file = here("models", "beta_reg", "beta_reg_prior_predictive_check")
)

beta_reg_group_stress <- brm(
  formula = beta_reg_formula, 
  prior = beta_reg_prior, 
  data = bi_4_prop, 
  cores = 4, backend = "cmdstanr", threads = threading(2),

bi_zoid_tib_prep <- bind_rows(
  summary(beta_reg_group_stress)[["fixed"]] %>% 
    as_tibble(rownames = "Term") %>% 
    mutate(Effect = "Population"),
  summary(beta_reg_group_stress)[["random"]][["id"]] %>% 
    as_tibble(rownames = "Term") %>% 
    mutate(Effect = "Grouping")
) %>% 
  mutate(across(-c("Effect", "Term"), specify_decimal, k = 2)) %>% 
  mutate(Term = case_when(
      Term == "phi_Intercept" ~ "ϕ Intercept",
      Term == "zi_Intercept" ~ "zi Intercept",
      Term == "group_sum" ~ "Group", 
      Term == "stress_sum" ~ "Stress", 
      Term == "prof_z" ~ "Proficiency", 
      Term == "use_z" ~ "Use", 
      Term == "group_sum:stress_sum" ~ "Group:Stress", 
      Term == "group_sum:use_z" ~ "Group:Use", 
      Term == "stress_sum:use_z" ~ "Stress:Use", 
      Term == "group_sum:prof_z" ~ "Group:Proficiency", 
      Term == "Stress_sum:prof_z" ~ "Stress:Proficiency", 
      Term == "use_z:prof_z" ~ "Use:Proficiency", 
      Term == "group_sum:stress_sum:use_z" ~ "Group:Stress:Use", 
      Term == "group_sum:stress_sum:prof_z" ~ "Group:Stress:Prof.", 
      Term == "group_sum:use_z:prof_z" ~ "Group:Use:Prof.", 
      Term == "stress_sum:use_z:prof_z" ~ "Stress:Use:Prof.", 
      Term == "group_sum:stress_sum:use_z:prof_z" ~ "Group:Stress:Use:Prof.", 
      Term == "sd(stress_sum)" ~ "sd(Stress)", 
      Term == "sd(phi_Intercept)" ~ "sd(ϕ Intercept)", 
      Term == "sd(zi_Intercept)" ~ "sd(zi Intercept", 
      Term == "cor(Intercept,stress_sum)" ~ "cor(Intercept,Stress)", 
      TRUE ~ Term)) 

bi_zoid_tib <- bi_zoid_tib_prep %>%
