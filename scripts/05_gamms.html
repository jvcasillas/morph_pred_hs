# Set up data frame
bi_50_gamm <- stress_50 %>% 
  filter(group != "Monolingual", 
    time_zero > -4, time_zero < 10) %>% 
  mutate(id = as.factor(id), 
    target = as.factor(target), 
    stress = as.ordered(stress),
